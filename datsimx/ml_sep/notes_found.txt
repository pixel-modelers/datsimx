# MONOCHROMATIC, MULTI-LATTICE PATTERN SEPERATION

# Original command for making mono sims
datsimx.sim mono_1ksims --numimg 1000 --cuda --mono --oversample 1 --mosDoms 1 --ndev 4 --randomizerot

# script used to make the training data file
mpirun  -n 64 python datsimx/mlat_combine.py  "mono_1ksims/proc/*strong.refl" mono_1ksims/train_masks.h5

# last script used to train
python datsimx/ml_sep/train.py  mono_1ksims/train_masks.h5 ml_test.h5 sep_mlatts3 --bs 12 --nwork 12 --lr 1e-2 --nep 1000 --goal sepMulti --savefreq 1

# POLYCHROMATIC PATTERN SEPARATION

# original command for making poly sims
datsimx.sim poly_sims --numimg 100 --cuda --oversample 1 --mosDoms 1 --ndev 4 --randomizerot

# script used to combine the wavelength data
mpirun -n 4 python datsimx/wave_combine.py  "poly_sims_test/proc/*strong.refl" poly_sims_test/test_masks.h5

# last script used to train
python datsimx/ml_sep/train.py  poly_sims/train_masks.h5 poly_sims_test/test_masks.h5 peakwave --bs 6  --nwork 6  --lr 10   --nep 1000 --goal peakWave --savefreq 10
